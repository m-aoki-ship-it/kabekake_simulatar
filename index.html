// ARWallMeasurementView.swift // 壁掛けテレビARシミュレーター // // 【重要】実際のプロジェクトに組み込む際は、 // Info.plist に NSCameraUsageDescription の記載が必要です。 import SwiftUI import ARKit import RealityKit import Combine // MARK: - メインのARビュー（SwiftUIから呼び出す） struct ARWallMeasurementView: UIViewRepresentable { @ObservedObject var coordinator: ARCoordinator func makeUIView(context: Context) -> ARView { let arView = ARView(frame: .zero) // LiDAR非搭載機でも動くようにワールドトラッキングを選択 let config = ARWorldTrackingConfiguration() config.planeDetection = [.vertical] // 壁だけを検出 arView.session.run(config) arView.session.delegate = context.coordinator // タップジェスチャーを追加 let tapGesture = UITapGestureRecognizer( target: context.coordinator, action: #selector(ARCoordinator.handleTap(_:)) ) arView.addGestureRecognizer(tapGesture) context.coordinator.arView = arView return arView } func updateUIView(_ uiView: ARView, context: Context) {} func makeCoordinator() -> ARCoordinator { return coordinator } } // MARK: - ARの状態とロジックを管理するコーディネーター class ARCoordinator: NSObject, ObservableObject, ARSessionDelegate { weak var arView: ARView! @Published var leftPoint: CGPoint? // 左端のタップ座標 @Published var rightPoint: CGPoint? // 右端のタップ座標 @Published var measuredWidth: Float = 0.0 // 実寸幅（メートル） @Published var statusMessage = "壁の左端をタップしてください" private var tvAnchor: AnchorEntity? @objc func handleTap(_ sender: UITapGestureRecognizer) { let location = sender.location(in: arView) // ********** ここが今回のキモ ********** // ARKit のレイキャストで、検出済みの「壁平面」との正確な交点を取得 let results = arView.raycast(from: location, allowing: .estimatedPlane, alignment: .vertical) guard let firstResult = results.first else { statusMessage = "壁を認識できません。もう少しゆっくりスマホを動かしてください" return } // ************************************ if leftPoint == nil { // 1回目: 左端として記録 leftPoint = location statusMessage = "壁の右端をタップしてください" // タップ位置に小さな目印を表示（オプション） placeMarker(at: firstResult, color: .green) } else if rightPoint == nil { // 2回目: 右端として記録し、計測開始 rightPoint = location placeMarker(at: firstResult, color: .green) // 右端の3D空間上の座標も取得 if let leftResult = arView.raycast(from: leftPoint!, allowing: .estimatedPlane, alignment: .vertical).first { // 2点間のユークリッド距離を計算 → これがテレビの「幅」になる let distance = simd_distance(leftResult.worldTransform.translation, firstResult.worldTransform.translation) measuredWidth = distance statusMessage = String(format: "幅: %.1f cm (タップで再測定)", distance * 100) // テレビの3Dモデルを正しいサイズで配置 placeTVModel(left: leftResult, right: firstResult) } } else { // 再測定: すべてリセット resetMeasurement() } } // 計測結果をリセット func resetMeasurement() { leftPoint = nil rightPoint = nil tvAnchor?.removeFromParent() tvAnchor = nil measuredWidth = 0.0 statusMessage = "壁の左端をタップしてください" } // タップ位置に目印を表示 private func placeMarker(at result: ARRaycastResult, color: UIColor) { let sphere = MeshResource.generateSphere(radius: 0.01) let material = SimpleMaterial(color: color, isMetallic: false) let entity = ModelEntity(mesh: sphere, materials: [material]) let anchor = AnchorEntity(world: result.worldTransform) anchor.addChild(entity) arView.scene.addAnchor(anchor) } // テレビの3Dモデルを正確なサイズで表示 private func placeTVModel(left: ARRaycastResult, right: ARRaycastResult) { // 古いモデルがあれば削除 tvAnchor?.removeFromParent() let width = simd_distance(left.worldTransform.translation, right.worldTransform.translation) // テレビの簡易3Dモデルを作成（薄い直方体） // 実際のアプリでは、USDZファイルを読み込むことを推奨 let tvThickness: Float = 0.05 // 5cmの厚み let tvHeight = width * 9.0 / 16.0 // 16:9のアスペクト比 let tvMesh = MeshResource.generateBox( width: width, height: tvHeight, depth: tvThickness ) // 真っ黒なテレビ画面のマテリアル var tvMaterial = PhysicallyBasedMaterial() tvMaterial.baseColor = .init(tint: .black) tvMaterial.roughness = 0.1 tvMaterial.metallic = 0.0 let tvEntity = ModelEntity(mesh: tvMesh, materials: [SimpleMaterial()]) // 2点の中間に配置 let midX = (left.worldTransform.translation.x + right.worldTransform.translation.x) / 2 let midY = (left.worldTransform.translation.y + right.worldTransform.translation.y) / 2 + tvHeight / 2 let midZ = (left.worldTransform.translation.z + right.worldTransform.translation.z) / 2 var transform = Transform() transform.translation = SIMD3(midX, midY, midZ) tvEntity.transform = transform tvAnchor = AnchorEntity() tvAnchor!.addChild(tvEntity) arView.scene.addAnchor(tvAnchor!) } } // MARK: - 座標変換のための拡張 extension simd_float4x4 { var translation: SIMD3 { return SIMD3(columns.3.x, columns.3.y, columns.3.z) } }
